111 virtual int select (int button) {
112
113 if (isOpened() && (button >= 0) && (button <= W)) {
114
115 for (unsigned int i = 0; i < m_spin.size(); i++) {
116
117 digitalWrite (m_spin.at (i), (button & (1 << i)) ? HIGH : LOW);

119 }

122 }
